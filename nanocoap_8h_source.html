549 static inline unsigned coap_get_token_len(const coap_pkt_t *pkt)
550 {
551  uint8_t tkl = pkt->hdr->ver_t_tkl & 0xf;
552 
553  if (!IS_USED(MODULE_NANOCOAP_TOKEN_EXT)) {
554  return tkl;
555  }
556 
557  void *ext = pkt->hdr + 1;
558  switch (tkl) {
559  case 13:
560  return tkl + *(uint8_t *)ext;
561  case 14:
562  return tkl + 255 + byteorder_bebuftohs(ext);
563  case 15:
564  assert(0);
565  /* fall-through */
566  default:
567  return tkl;
568  }
569 }

82 static inline uint32_t _get_next_rs_interval(const gnrc_netif_t *netif)
83 {
84  if (gnrc_netif_is_6ln(netif)) {
85  if (netif->ipv6.rs_sent < SIXLOWPAN_ND_MAX_RS_NUMOF) {
86  return SIXLOWPAN_ND_RS_MSEC_INTERVAL;
87  }
88  else {
89  unsigned exp = netif->ipv6.rs_sent - SIXLOWPAN_ND_MAX_RS_NUMOF;
90  uint32_t tmp = SIXLOWPAN_ND_RS_MSEC_INTERVAL +
91  ((1 << exp) * (NDP_RS_MS_INTERVAL));
92 
93  if (tmp > (SIXLOWPAN_ND_MAX_RS_SEC_INTERVAL * MS_PER_SEC)) {
94  tmp = SIXLOWPAN_ND_MAX_RS_SEC_INTERVAL * MS_PER_SEC;
95  }
96  return tmp;
97  }
98  }
99  else {
100  return NDP_RS_MS_INTERVAL;
101  }
102 }
